<!DOCTYPE html>
<html lang="zh-tw">
<!--整份網頁文件，用來包<head>、<body>標籤-->   
    <head>
        <title>網頁標題</title>
        <style>css樣式
        <meta>提供搜尋引擎的重要資訊
        <!--
        charset="UTF-8"
        網頁編碼

        http-equiv="X-UA-Compatible" content="IE=edge" 
        自動將IE瀏覽器升級

        http-equiv="refresh" content="5; url="轉跳網址"
        進入該網站後5秒後，自動轉跳至其他網頁

        name="viewport" content="width=device-width, initial-scale=1.0",(user-scalable=no 禁止使用者縮放)
        響應手機版，首次載入頁面的縮放比 
        
        name="description" content="網頁說明"
        內容建議以，分隔

        name="keywords" content="網頁關鍵字"
        google不看，可改放在說明

        name="author" content="網頁作者"

        name="robots" content="noindex"搜尋引擎不要索引這個網頁
        

        property="og:title" content="顯示文字"
        property="og:image" content="顯示的圖片網址"
        property="og:type" content="website"類型
        property="og:description" content="描述內容文字"
        -->
        <link rel="stylesheet" href="style.css">引入檔案連結
        <base target="_blank">定義該網頁的連結方式
        <script src="引入js檔案連結"></script>
    </head>
    <body>
        <script src="引入js檔案連結"></script>
        <header>
        <nav>
        <section>用來包一段文章<h*>、<p>標籤
        <article>用來包<section>標籤
        <aside>
        <footer>
        <details>
        <summary>
    </body>
</html>
