//取出"字串"資料，並轉成JSON格式
let getData = localStorage.getItem("fruitName");
let fruitArray = JSON.parse(getData);

let todoData = [];
let newTodo = document.getElementById("newTodo");
let addTodo = document.getElementById("addTodo");
let todoList = document.getElementById("todoList");

//初始化資料
function initData(){
    //取得localStorage上的"todoItem"資料
    let getData = JSON.parse(localStorage.getItem("todoItem"));
    //將"字串"資料轉為JSON格式
    let todoArray = JSON.parse(getData);
    //將原本是空資料賦予取得的資料
    todoData = todoArray;
}
initData();

//更新資料
function updateList(){
    //建立空字串
    let str = "";
    //將所有資料跑一遍
    for(let i=0 ; i<todoData.length ; i++){
        //並組成<li>並且在<a>連結裡面帶入data-num編號
        str += "<li><a href='#' data-num='"+i+"'>刪除</a>"+todoData[i]+"</li>";
    }
    //組裝完成後，渲染到頁面上
    todoList.innerHTML = str;
}
updateList();

//新增資料
addTodo.addEventListener("click",function(){
    //如果input裡有值才執行
    if(newTodo.value !== ""){
        //將input裡的值新增到資料
        todoData.push(newTodo.value);
        //將整個資料轉成"字串"
        let setData = JSON.stringify(todoData);
        //儲存到localStorage上
        localStorage.setItem("todoItem",setData);
        //重新渲染頁面的資料
        updateList();
        //清空input裡的值
        newTodo.value = "";
    }
},false)

//刪除資料
todoList.addEventListener("click",function(e){
    //當我點擊的是ul裡的a才執行
    if(e.target.nodeName == "A"){
        //取得我點擊目標的data-num編號
        let targetNum = e.target.dataset.num;
        //並從資料中刪除該項
        todoData.splice(targetNum,1);
        //將整個資料轉成"字串"
        let setData = JSON.stringify(todoData);
        //儲存到localStorage上
        localStorage.setItem("todoItem",setData);
        //重新渲染頁面的資料
        updateList();
    }
})
