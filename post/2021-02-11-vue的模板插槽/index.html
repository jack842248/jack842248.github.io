// <-----html----->
#app
  p-cp 
    li 香蕉
  
//．蘋果

// <-----js----->
var app = new Vue({
  el: "#app",
  components:{
    "p-cp":{
      template: 
      `
      <ul>
        <li>蘋果</li>
      </ul>
      `
    }
  }
});

  • 使用<template>自己標籤不被輸出，所以不會覆蓋標籤。

  • <slot>可以讓"蘋果"被替換，若沒有被替換，“蘋果"也會變成預設值而顯示出來。

  components:{
    "p-cp":{
      template:

      <ul>
        <li>
          <slot>蘋果</slot>
        </li>
      </ul>
      `
    }
  }
});

  • 使用name=“自訂名稱”，可對應solt=“自訂名稱”。

  el: "#app",
  components:{
    "p-cp":{
      template: 
      `
      <ul>
        <li>蘋果</li>
        <slot name="a">蘋果</slot>
        <li>蘋果</li>
      </ul>
      `
    }
  }
});
