.text{
    color:red/#333333/rgba(3,3,3,0); /*顏色名稱/色票/最後的值調整透明度*/
    font-size:large/16px/1em; /*文字大小*/
    font-family:"微軟正黑體","import"; /*優先使用前者*/
    font-weight:100/900/bold; /*字體粗度*/
    font-style:normal/italic; /*正常/斜體*/
    font-variant:small-caps; /*小型的大寫字體*/
    text-align:left/right/center/justify; /*靠左/靠右/置中/左右對齊*/
    text-indent:2em; /*縮排2個字*/
    text-decoration:underline/overline/line-through; /*上線條/下線條/刪除線*/
    text-transform:capitalize/uppercase/lowercase; /*每個單字開頭大寫/所有字母大寫/所有字母小寫*/
    text-shadow:30px 40px; /*X軸 Y軸*/
    text-shadow:30px 40px red; /*X軸 Y軸 陰影顏色*/
    text-shadow:30px 40px 20px red; /*X軸 Y軸 模糊 陰影顏色*/
    text-shadow:30px 40px 20px 10px red; /*X軸 Y軸 模糊 擴散 陰影顏色*/
    text-shadow:30px 40px red,10px 50px blue; /*前者在上層，後者在下層*/
    word-spacing:5px; /*每個單字間的距離*/
    letter-spacing:8px; /*文字間距*/
    line-height:40px; /*段落行距*/
    white-space:nowrap/normal; /*不換行/自動換行*/
    word-break:keep-all/break-all; /*換行時單字不會被切/換行時單字會被切*/
    unicode-bidi:bidi-overide; /*文字相反顯示*/
    writing-mode:horizontal-tb/vertical-rl; /*文字水平/文字垂直*/
}

.box{
    background-color:red/#333333/rgba(3,3,3,0); /*顏色名稱/色票/最後的值調整透明度*/
    background-size:200px/50%/cover/contain; /*200px大小/縮小百分之50/填滿元素大小/圖片大於元素時能完整呈現*/
    background-position:right top/20px 30px; /*在右上方/X軸 Y軸*/
    background-repeat:repeat/no-repeat/repeat-x/repeat-y; /*重複/不重複/X軸重複/Y軸重複*/
    background-attachment:scroll/fixed; /*背景捲動時跟著移動/背景固定不動*/
    background-image:url("連結網址"); /*崁入圖片*/
    background-image:linear-gradient(red,yellow); /*上到下，紅到黃*/
    background-image:linear-gradient(45deg,red,yellow); /*右上到左下，紅到黃*/
    background-image:linear-gradient(90deg,red,transparent); /*右到左，紅到透明*/
    background-image:radial-gradient(red,blue); /*由內到外，紅到藍*/
    background-image:radial-gradient(red 5%,blue 10%); /*由內到外，紅5%到藍10%*/
    background-image:radial-gradient(circle,red,blue); /*由內到外(正圓)，紅到藍*/
    background-image:radial-gradient(repeating,circle,red,blue); /*重複由內到外(正圓)，紅到藍*/

    opacity:0.0/1.0/inherit; /*透明度*/
}

    border-width:5px/thin/medium/thick; /*線條寬度/薄/中/厚*/

    border-left:solid 1px red; /*左邊框線樣式/框線寬度/框線顏色*/
    border-radius:30px/100%; /*邊框圓角/圓型*/

    overflow:visible/hidden/scroll/auto; /*讓內容超出範圍/讓內容超出範圍就隱藏/在超出的內容加上捲軸/在超出的內容的軸向加上捲軸*/
    overflow-x:hidden/scroll; /*只針對X軸向*/
    overflow-y:hidden/scroll; /*只針對Y軸向*/
    resize:none/both/horizontal/vertical/block; /*不增加窗角/可任意拉動/可水平拉動/水垂直拉動*/
}

.box{
    position:static/relative/absolute/fixed/sticky;
    /*
    static靜態:自動排列下來，不能偏移位置
    relative相對定位:自動排列下來，可以偏移位置
    absolute絕對定位:對上層物件作做偏移定位(但上層一定要是relative)
    fixed固定:固定在某個位置，不會隨著視窗滾動
    sticky凍結在上方:視窗向下滾動，仍停留在上方
    */
}

.box{
    display:block/inline/inline-block/flex/none;
    /*
    block區塊元素:<div>、<p>、<form>、<hr>、<dl>、<dt>、<dd>、<ol>、<ul>、<h1>、<h6>
    inline行內元素:<span>、<a>、<small>、<label>
    inline-block行內元素並包覆:<img>、<input>、<select>、<button>
    flex:彈性自由排列
    none:不顯示
    */

    visibility:visible/hidden/collapse; /*正常顯示/隱藏(區塊保留)/不顯示*/
}

.box{
    transform:translate(30px,20px); /*偏移X軸30px Y軸20px*/
    transform:translateX(30px); /*偏移X軸30px*/
    transform:translateY(20px); /*偏移Y軸20px*/
    transform:scale(1.5,2); /*縮放X軸1.5 Y軸2*/
    transform:scaleX(1.5); /*縮放X軸1.5*/
    transform:scaleY(2); /*縮放Y軸2*/
    transform:skew(30deg,20deg); /*傾斜X軸30deg*/
    transform:skewX(30deg); /*傾斜X軸30deg*/
    transform:skewY(20deg); /*傾斜Y軸20deg*/
    transform:rotate(30deg); /*旋轉30度*/
    /*正值為順時針，負值為逆時針*/
    transform-origin:center/50px 20px; /*原點在中心/從原點離上方50px 離左方20px*/
}
