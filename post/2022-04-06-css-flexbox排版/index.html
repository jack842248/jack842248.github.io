ul{
    display: flex;

    flex-direction: row; /*左到右*/
    flex-direction: row-reverse; /*右到左*/
    flex-direction: column; /*上到下*/
    flex-direction: column-reverse; /*下到上*/

    justify-content: flex-start; /*從頭開始*/
    justify-content: flex-end; /*從尾開始*/
    justify-content: center; /*置中*/
    justify-content: space-between; /*貼齊左右兩側*/
    justify-content: space-around; /*平均分散*/

    align-items: flex-start; /*從頭開始*/
    align-items: flex-end; /*從尾開始*/
    align-items: center; /*置中*/
    align-items: stretch; /*延伸到底*/
    align-items: baseline; /*對齊基線*/

    flex-wrap: nowrap; /*不換行*/
    flex-wrap: wrap; /*換行*/
    flex-wrap: wrap-reverse; /*換行相反*/
}
li{
    align-content: flex-start; /*對齊頭*/
    align-content: flex-end; /*對齊尾*/
    align-content: center; /*對齊中線*/
    align-content: space-between; /*第一行對齊頭，最後一行對齊尾*/
    align-content: space-around; /*平均分配*/
    align-content: stretch; /*延伸到底*/

    flex-basis: 200px; /*設定該元素大小*/

    flex-shrink: 0; /*設定該元素彈性*/

    flex-grow: 2; /*設定該元素佔比*/

    flex: none; /*保持原尺寸*/
    flex: 3; /*永遠佔30%*/

    order: 1~100; /*設定排序，數字小的先*/
}
