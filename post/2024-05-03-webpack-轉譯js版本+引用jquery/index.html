轉譯js

  1. 安裝套件babel-loader和@babel/core和@babel/preset-env套件
    • babel-loader是webpack的loader套件
    • @babel/core調用babel的api編譯
    • @babel/preset-env可以使用最新版本的javascript去編譯

$ npm i babel-loader @babel/core @babel/preset-env -D

  1. 在webpack.config.js新增：

const path = require('path');

module.exports = {
    entry: './src/js/main.js',
    output: {
        path: path.resolve(__dirname,'dist'),
        filename: './js/main.js', 
        clean: true, 
    },
    module: {
        rules: [
            {
                test: /\.js$/, //找尋js副檔名的檔案(正規表達式)
                exclude: /node_modules/,
                use: {
                    loader: 'babel-loader',
                    options: {
                        presets:['@babel/preset-env']
                    },
                }
            }
        ]
    }
}

  1. 在src底下新增all.js檔，並在檔案裡隨便寫一些程式：

    return `Hello, ${name}!`;

$ npm i jquery

const webpack = require('webpack'); //引用webpack-cli內的工具
const path = require('path');

module.exports = {
    entry: './src/js/main.js',
    output: {
        path: path.resolve(__dirname,'dist'),
        filename: './js/main.js', 
        clean: true, 
    },
    module: {
        rules: [
            {
                test: /\.js$/, //找尋js副檔名的檔案(正規表達式)
                exclude: /node_modules/,
                use: {
                    loader: 'babel-loader',
                    options: {
                        presets:['@babel/preset-env']
                    },
                }
            }
        ]
    },
    plugins: [
        new webpack.ProvidePlugin({ //全域設置$變數
            $: 'jquery',
            jquery: 'jquery',
            'window.jquery': 'jquery'
        })
    ]
}

  1. 測試jquery有沒有載入成功，可以在先前的all.js：

1
2
3

$(document).ready(function(){
    $("body").css('backgroundColor','red')
})

  1. 再次終端機執行：

$ npm run start

  1. 如果頁面整個有成功變成紅色，表示成功。
