li{
    list-style-type:disc/circle/square/upper-latin/lower-latin/upper-roman/lower-roman/none;
    /*黑圓點/黑圓圈/黑方塊/大寫英文/小寫英文/大寫羅馬文/小寫羅馬文/不顯示*/
    list-style-image:url("連結網址"); /*崁入清單樣式*/
    list-style-position:outside/inside; /*圖樣在元素外/圖樣在元素內*/
    list-style:circle inside; /*清單樣式/圖樣位置*/

    &::before{ /*設定在元素之前*/
        display:block/inline/inline-block;
        content:""; /*可輸入UTF-8字符或文字或留白*/
        position:relative/absolute;
    }

    &::after{ /*設定在元素之後*/
        display:block/inline/inline-block;
        content:""; /*可輸入UTF-8字符或文字或留白*/
        position:relative/absolute;
    }

    /*載入Font Awesome icon*/
    &::before{ 
        font-family:"Font Awesome 5 Free";
        content:"\f0e0";
        font-weight:"900";
        font-size:24px;
        position:relative/absolute;
        z-index:20;
    }
}

img{
    width:50px; /*設定寬度*/
    height:30px; /*設定高度*/
    max-width:200px; /*設定最大寬度*/
    max-height:100px; /*設定最大高度*/
    min-width:50px; /*設定最小寬度*/
    min-height:30px; /*設定最小高度*/

    /*自適應圖片*/
    width:100%;
    height:auto;

    /*圖片被文字圍繞*/
    float:left/right; /*浮動位置*/
    clear:left/right/both; /*左浮動取消/右浮動取消/左右都取消*/
    /*
    1.若該元素的position:static/position則無效
    2.在img上設定浮動，文字會圍繞在旁邊
    3.浮動的img可用padding/margin調整
    4.若要取消浮動在浮動的最後元後後面加上.clearfix
    */
}

.box{
    vertical-align:baseline/top/middle/bottom/30px/sub/super /*預設值/切齊上方/切齊中線/切齊下方/偏移中線30px/偏移向下/偏移向上*/
    /*
    1.限用同行的display:inline/inline-block元素<div><span><img>
    2.同行的元素都要加上vertical-align才會對齊
    3.不適用position:relative/absolute元素
    */
    
    /*文字垂直水平置中*/
    text-align:center;
    line-height:設跟height值一樣;

    /*元素水平置中*/
    margin:0 auto;

    /*元素垂直水平置中*/
    上層元素position:relative;
    position:absolute;
    left:50%;
    top:50%;
    transform:translate(-50%,-50%);

    /*文字/元素垂直水平置中*/
    上層元素display:flex;
    justify-content:center;
    align-items:center;
}
